  \def\side{#2}

    \draw[line width=1mm]
      ($#1$)
        -- ++(0,\side/2) 
        -- ++(\cornerrad,\cornerrad) -- ++(\side,0) 
        -- ++(\cornerrad,-\cornerrad) -- ++(0,-\side)
        -- ++(-\cornerrad,-\cornerrad) -- ++(-\side,0)
        -- ++(-\cornerrad,\cornerrad) -- cycle;
  }{%
    \draw[line width=#4,fill stretch image=#5]
      ($#1$)
        -- ++(0,\side/2)
